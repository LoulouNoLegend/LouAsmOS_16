Color Codes:
    0x0     Black
    0x1     Blue
    0x2     Green
    0x3     Cyan
    0x4     Red
    0x5     Magenta
    0x6     Brown

    0x7     Light Gray
    0x8     Dark Gray
    0x9     Light Blue
    0xA     Light Green
    0xB     Light Cyan
    0xC     Light Red
    0xD     Light Magenta

    0xE     Yellow
    0xF     White

AH = 09h
AL = character to print
BH = page number (0)
BL = color attribute byte
CX = repeat count
INT 10h
